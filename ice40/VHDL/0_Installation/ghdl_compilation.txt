first install verilog synthesys tools as default

than:

git clone https://github.com/ghdl/ghdl.git ~/ghdl
cd ghdl
mkdir build
cd build
sudo apt install gnat gprbuild

# Download gcc-srcfiles (whole folder) from https://mirror.sergal.org/gnu/gcc/gcc-9.5.0/ into ghdl/gccsrc
wget https://mirror.sergal.org/gnu/gcc/gcc-9.5.0/gcc-9.5.0.tar.gz
tar xvf gcc-9.5.0.tar.gz
rm -rf gcc-9.5.0.tar.gz
 -> copy into into ghdl/gccsrc

# use full path TODO: Try at with ${USER}
../configure --with-gcc=/home/jonas/ghdl/gccsrc/gcc-9.5.0 --prefix=/usr/local
make copy-sources
mkdir gcc-objs; cd gcc-objs

###### in second terminal:
# install gmp
wget https://gmplib.org/download/gmp/gmp-6.1.2.tar.xz
tar xvf gmp-6.1.2.tar.xz
rm -rf gmp-6.1.2.tar.xz
cd gmp-6.1.2
./configure --prefix=/usr/local/gmp/6_1_2
make
make check
sudo make install

# install mpfr
wget https://gcc.gnu.org/pub/gcc/infrastructure/mpfr-4.1.0.tar.bz2
bunzip2 mpfr-4.1.0.tar.bz2
tar xvf mpfr-4.1.0.tar
rm -rf mpfr-4.1.0.tar
cd mpfr-4.1.0
./configure --disable-shared --enable-static --prefix=/usr/local/mpfr/4_1_0 --with-gmp=/usr/local/gmp/6_1_2
make
make check
sudo make install

# install mpc
wget https://gcc.gnu.org/pub/gcc/infrastructure/mpc-1.2.1.tar.gz
tar zxvf mpc-1.2.1.tar.gz
cd mpc-1.2.1
./configure --disable-shared --enable-static --prefix=/usr/local/mpc/1_2_1 --with-gmp=/usr/local/gmp/6_1_2 --with-mpfr=/usr/local/mpfr/4_1_0
make
make check
sudo make install

# close second terminal


####### going on with ghdl

--with-mpfr=/usr/local/mpfr/4_1_0 --with-mpc=/usr/local/mpc/1_2_1
make -j2
sudo make install
